% Site-level OG image: both book covers side by side.
%
% Uses the already-rendered front-cover PNGs from each book's
% marketing/ directory.  Rebuild covers first if they've changed
% (cd book1 && make cover-assets; cd book2 && make cover-assets).
%
% Build: python3 site/scripts/build_og_image.py  (from repo root)

\documentclass[border=0pt]{standalone}
\usepackage[T1]{fontenc}
\usepackage{tikz}
\usepackage{graphicx}
\usepackage[default]{sourcesanspro}

% Shared palette
\definecolor{paper}{HTML}{F2EFE9}
\definecolor{muted}{HTML}{B8B5B0}

\begin{document}
\begin{tikzpicture}
    % Canvas: 24cm x 12.6cm  (1200:630 = 40:21 aspect ratio)
    \useasboundingbox (0,0) rectangle (24, 12.6);
    \fill[paper] (0,0) rectangle (24, 12.6);

    % --- Cover layout ---
    \pgfmathsetmacro{\coverH}{10.2}               % cover height (cm)
    \pgfmathsetmacro{\coverW}{\coverH * 2 / 3}    % 6:9 aspect ratio
    \pgfmathsetmacro{\gap}{1.4}                    % gap between covers
    \pgfmathsetmacro{\totalW}{2*\coverW + \gap}
    \pgfmathsetmacro{\startX}{(24 - \totalW) / 2}
    \pgfmathsetmacro{\coverY}{1.6}                 % bottom of covers

    % --- Drop shadows ---
    \fill[black, opacity=0.06]
        ({\startX+0.12}, {\coverY-0.12})
        rectangle +({\coverW}, {\coverH});
    \fill[black, opacity=0.06]
        ({\startX+\coverW+\gap+0.12}, {\coverY-0.12})
        rectangle +({\coverW}, {\coverH});

    % --- Boardroom cover (left) ---
    \node[inner sep=0, anchor=south west] at ({\startX}, {\coverY}) {
        \includegraphics[height=\coverH cm]{../../book2/marketing/front-cover.png}
    };

    % --- Work cover (right) ---
    \node[inner sep=0, anchor=south west] at ({\startX+\coverW+\gap}, {\coverY}) {
        \includegraphics[height=\coverH cm]{../../book1/marketing/front-cover.png}
    };

    % --- URL ---
    \node[muted, font=\sffamily\footnotesize] at (12, 0.6)
        {simpleisadvanced.com};
\end{tikzpicture}
\end{document}
